### git命令总结

```bash
git init
git add file1 file2
git commit -m "message"
git log
git status
git reset --hard 版本号
git reflog
git checkout --file
git reset HEAD file
git branch dev
git switch dev
git branch -d dev
git merge dev
git remote add origin git@github.com   .git
git push -u origin master
git push origin dev:master
git clone git@github.com  .git
git fetch

```
